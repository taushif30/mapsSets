Q2 -Problem: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps) Input: abcadeecfb Output: a=2 b=2 c=2 d=1 e=2 f=1


    ANS:function count(str){
        const alphabetCount={}
    
    
        for(let char of str ){
            if(char in alphabetCount)
            alphabetCount[char]++
        
        else{
            alphabetCount[char]=1
        }
        }
        for (let char in alphabetCount){
            console.log(`${char} = ${alphabetCount[char]}`)
        }
    
    }
    
    count("abcadeecfb")

    OUTPUT : a=2
            b=2
            c=2
            d=1
            e=2
            f=1
